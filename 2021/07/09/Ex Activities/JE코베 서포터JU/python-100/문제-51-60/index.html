문제 51) merge sort를 만들어보자

def merge_sort(lst):
    len_lst = len(lst)
    if len_lst <= 1:
        return lst
    mid = len_lst // 2
    g1 = merge_sort(lst[:mid])
    g2 = merge_sort(lst[mid:])
    res = []

    while g1 and g2 :
        if g1[0] < g2[0]:
            res.append(g1.pop(0))
        else:
            res.append(g2.pop(0))

    while g1:
        res.append(g1.pop(0))
    while g2:
        res.append(g2.pop(0))
    return res

lst = [180, 145, 165, 45, 170, 175, 173, 171]
print(merge_sort(lst))

문제 52) 문자열 만들기

def quick_sort(lst):
    len_lst = len(lst)
    if len_lst <= 1:
        return lst
    mid = lst.pop(len_lst//2)
    g1 = []
    g2 = []
    
    for i in range(len_lst-1):
        if lst[i] < mid:
            g1.append(lst[i])
        else:
            g2.append(lst[i])
    return quick_sort(g1) + [mid] + quick_sort(g2)

lst = input().split(' ')
lst = [int(i) for i in lst]

print(quick_sort(lst))

문제 53) 괄호 문자열

def chk_parenthesis(s):
    if s.count('(') != s.count(')'):
        return False
    bracket = []
    for i in s:
        if i == '(':
            bracket.append('(')
        if i == ')':
            if len(bracket) == 0:
                return False
            bracket.pop()
    return True

n = input()
if chk_parenthesis(n):
    print("YES")
else:
    print("NO")

stack을 이용하여 괄호 짝을 맞춰주면 된다.

문제 54) 연속되는 수

def chk_sequence(lst):
    lst = sorted(lst)
    for i in range(len(lst) - 1):
        if lst[i] + 1 != lst[i + 1]:
            return "NO"
    return "YES"

lst = list(map(int, input().split()))
print(chk_sequence(lst))

문제 55) 하노이의 탑

path = []
def hanoi(n, src, dst, spt):
    #원판이 한개일 때에는 옮기면 됩니다.
    if n == 1:
        path.append([src, dst]) 
        return None
    #원반의 n-1개를 경유기둥으로 옮기고
    hanoi(n-1, src, spt, dst)
    #가장 큰 원반은 목표기둥으로
    path.append([src, dst]) 
    #경유기둥과 시작기둥을 바꿉니다!
    hanoi(n-1, spt, dst, src)

user_input = int(input())
hanoi(user_input,'A','C','B') # 원반의수, 시작기둥, 목표기둥, 보조기둥

print(len(path))
