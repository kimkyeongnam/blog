def merge_sort(lst):
    len_lst = len(lst)
    if len_lst <= 1:
        return lst
    mid = len_lst // 2
    g1 = merge_sort(lst[:mid])
    g2 = merge_sort(lst[mid:])
    res = []

    while g1 and g2 :
        if g1[0] < g2[0]:
            res.append(g1.pop(0))
        else:
            res.append(g2.pop(0))

    while g1:
        res.append(g1.pop(0))
    while g2:
        res.append(g2.pop(0))
    return res

lst = [180, 145, 165, 45, 170, 175, 173, 171]
print(merge_sort(lst))

def quick_sort(lst):
    len_lst = len(lst)
    if len_lst <= 1:
        return lst
    mid = lst.pop(len_lst//2)
    g1 = []
    g2 = []
    
    for i in range(len_lst-1):
        if lst[i] < mid:
            g1.append(lst[i])
        else:
            g2.append(lst[i])
    return quick_sort(g1) + [mid] + quick_sort(g2)

lst = input().split(' ')
lst = [int(i) for i in lst]

print(quick_sort(lst))

def chk_parenthesis(s):
    if s.count('(') != s.count(')'):
        return False
    bracket = []
    for i in s:
        if i == '(':
            bracket.append('(')
        if i == ')':
            if len(bracket) == 0:
                return False
            bracket.pop()
    return True

n = input()
if chk_parenthesis(n):
    print("YES")
else:
    print("NO")

def chk_sequence(lst):
    lst = sorted(lst)
    for i in range(len(lst) - 1):
        if lst[i] + 1 != lst[i + 1]:
            return "NO"
    return "YES"

lst = list(map(int, input().split()))
