	i = 2
	while i < n:
		if n % i == 0:
			break
		i += 1
	if i == n:
		print("YES")
	else:
		print("NO")

def bubble(n, data):
	for i in range(n-1):
		for j in range(n-1-i):
			if data[j] > data[j+1]:
				data[j], data[j+1] = data[j+1], data[j]
	for i in range(n):
		print(data[i], end = " ")

n = int(input())
data = list(map(int, input().split()))

bubble(n, data)
