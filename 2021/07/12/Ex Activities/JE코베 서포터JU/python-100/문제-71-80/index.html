문제 71) 깊이 우선 탐색

graph = {
    'A': set(['B', 'C', 'E']),
    'B': set(['A']),
    'C': set(['A']),
    'D': set(['E', 'F']),
    'E': set(['A', 'D']),
    'F': set(['D'])
}

def dfs(start):
    visited = []
    stack = [start]

    while stack:
        n = stack.pop()
        if n not in visited:
            visited.append(n) 
            stack += graph[n] - set(visited)
    return visited

print(dfs('E'))

문제 72) 너비 우선 탐색

graph = {
    'A': set(['B', 'C', 'E']),
    'B': set(['A']),
    'C': set(['A']),
    'D': set(['E', 'F']),
    'E': set(['A', 'D']),
    'F': set(['D'])
}

def bfs(start):
    visited = []
    queue = [start]

    while queue:
        n = queue.pop(0)
        if n not in visited:
            visited.append(n)
            queue += graph[n] - set(visited)
    return visited

print(bfs('E'))

문제 73) 최단 경로 찾기

graph = {
    'A': set(['B', 'C']),
    'B': set(['A', 'D', 'E']),
    'C': set(['A', 'F']),
    'D': set(['B']),
    'E': set(['B', 'F']),
    'F': set(['C', 'E'])
}

def solution():
    start, end = [i for i in input().split()]
    
    queue = [start]
    visited = [start]

    cnt = -1
    while len(queue)!=0:
        cnt += 1
        size = len(queue)

        for i in range(size):
            node = queue.pop(0)
            if node == end:
                return cnt

            for next_node in graph[node]:
                if next_node not in visited:
                    visited.append(next_node)
                    queue.append(next_node)

print(solution())

문제 74) 최장 경로 찾기

graph = {
    1: [2, 3, 4],
	2: [1, 3, 4, 5, 6],
	3: [1, 2, 7],
	4: [1, 2, 5, 6],
	5: [2, 4, 6, 7],
	6: [2, 4, 5, 7],
	7: [3, 5, 6]
}

start, end = [int(i) for i in input().split()]
queue = [start]
visited = []

def solution(n, visited):
    if n[-1] == end:
        return len(visited)
    if n[-1] in visited:
        return len(visited)

    visited.append(n[-1])
    length = 0

    for next_node in graph[n[-1]]:
        n.append(next_node)
        length = max(length, solution(n, visited))
        queue.pop(0)
    return length

print(solution(queue, visited))

문제 77) 가장 긴 공통 부분 문자열

def solution(s):
    res = []
    for i in range(1, len(s) - 1):
        for j in range(i):
            res.append(s[j:j+len(s)-i+1])
    return res

input1 = input()
input2 = input()

# 문자열 나열될 수 있는 모든 경우의 수 만들기
lst1 = set(solution(input1))
lst2 = set(solution(input2))

# 경우의 수 교집합
ans = lst1.intersection(lst2)

# 가장 긴 교집합
ans = max(ans, key=len)
print(len(ans))

문제 78) 원형테이블

def solution(n, k):
    q = [i for i in range(1, n+1)]
    i = 0

    while len(q) > 2:
        if i > len(q)-1:
        # 순환하다 i가 q의 길이보다 클 경우에 len(q)만큼 빼준다.
        # [1,2,3,4,5,6] -> 1다음 4가 빠지고 그 다음은 4+3 = 7(index : 6)이 빠져야 하는데 
        # i(현재 빠져야 할 index)가 len(q)보다 크므로 len(q)즉, 4를 빼준다. 
        # 이걸 마지막 2개가 남을 때 까지 반복함
            i -= len(q)
        q.pop(i)
        i += k
        i -= 1
    print(q)

n, k = input().split()
solution(int(n), int(k))
