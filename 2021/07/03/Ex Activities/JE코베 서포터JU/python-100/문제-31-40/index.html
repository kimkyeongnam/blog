문제 37) count 사용하기

data = list(map(str, input().split()))
cnt = 0
for i in range(len(data)):
	if data.count(data[i-1]) < data.count(data[i]):
		cnt = i
print("{}(이)가 총 {}표로 반장이 되었습니다.".format(data[cnt], data.count(data[cnt])))

count = 0
