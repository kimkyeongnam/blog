문제 66) 블럭탑쌓기

def solution(tower, rule):
    answer = []
    for t in tower:
        answer.append(chk_rule(t, rule))
    return answer

def chk_rule(tower, rule):
    tmp = rule.index(rule[0])
    for c in tower:
        if c in rule:
            if tmp > rule.index(c):
                return '불가능'
            tmp = rule.index(c)
    return '가능'

tower = ['ABCDEF', 'BCAD', 'ADEFQRX', 'BEDFG', 'AEBFDGCH']
rule = 'ABCD'

print(solution(tower, rule))

문제 67) 민규의 악수

def solution(n):
    people = 0  # 행사참가자
    total = 0   # 악수횟수
    while True:
        total = (people * (people - 1)) // 2
        if n < total:
            break
        people += 1
    times = int(people - (total - n) - 1)
    return [times, people]
  • [a, b, c, d]가 악수를 했을 때:
    • [a, b]
    • [a, c]
    • [a, d]
    • [b, c]
    • [b, d]
    • [c, d]
    • 총 악수 횟수: 3 x 2 x 1 = 6
  • [1, 2, 3, …, n]가 악수를 했을 때:
    • 1: n - 1번의 악수
    • 2: n - 2번의 악수
    • n: 1번의 악수
    • 총 악수 횟수: (n - 1) + (n - 2) + … + 1 = (n * (n - 1)) // 2

문제 68) 버스 시간표

# 답 1
def sol(tb, rt):
    answer = []
    rt = rt.split(':')
    for i in range(len(tb)):
        time = tb[i].split(':')
        time_to_min = ((int(time[0])*60 + int(time[1])) - (int(rt[0])*60+int(rt[1])))
        if time_to_min < 0:
            answer.append("지나갔습니다")
        else:
            a = (time_to_min) // 60
            b = (time_to_min) % 60
            answer.append(str(a).zfill(2)+'시간 '+str(b).zfill(2)+'분')
    return answer

sol(["12:30", "13:20", "14:13"], "12:40")



# 답 2
def solution(lst, tgt):
    def generate_num(t):
        time = t.split(':')
        return int(time[0] + time[1])
    target = generate_num(tgt)
    for time in lst:
        gap = generate_num(time) - target
        if gap < 0:
            print("지나갔습니다")
        else:
            hour = gap // 60
            mins = gap - 60 * hour
            print(hour + "시간 " + mins + "분")

user_input = input().split()
tgt_time = input()

print(solution(user_input, tgt_time))

  • zfill로 자릿수를 맞춤

def cal():
    n=10000*2
    primes=[]    
    a = [False,False] + [True]*(n-1)

    for i in range(2,n+1):
      if a[i]:
        primes.append(i)
        for j in range(2*i, n+1, i):
            a[j] = False
            
    return primes

문제 70) 행렬 곱하기

def sol(a, b):
    c = []
    if len(a) == len(b[0]):
        for i in range(len(a)):
            c.append([0]*len(b[0]))
        for i in range(len(c)):
            for j in range(len(c[i])):
                for k in range(len(a[i])):
                    c[i][j] += a[i][k] * b[k][j]
        return c
    else:
        return -1
